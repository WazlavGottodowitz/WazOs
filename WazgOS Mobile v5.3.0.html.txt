  <script>
    const systemLog = (message) => {
      const container = document.getElementById('log');
      container.innerHTML += `<div>> ${message}</div>`;
      container.scrollTop = container.scrollHeight;
    };

    const TabRegistry = ['core', 'rig', 'motion', 'fx', 'oracle'];
    let activeIndex = 0;
    let base64Frame1Image = null;

    function navigateToTab(targetId) {
      activeIndex = TabRegistry.indexOf(targetId);
      document.querySelectorAll('.plugin-panel').forEach(panel => panel.classList.remove('active'));
      document.getElementById(`tab-${targetId}`).classList.add('active');
      document.querySelectorAll('.nav-btn').forEach(btn => btn.classList.remove('active'));
      document.getElementById(`btn-${targetId}`).classList.add('active');
    }

    TabRegistry.forEach(id => {
      document.getElementById(`btn-${id}`).addEventListener('click', () => navigateToTab(id));
    });

    document.getElementById('timeline').addEventListener('input', (event) => {
      document.getElementById('frame-num').innerText = event.target.value;
    });

    document.getElementById('frame1-uploader').addEventListener('change', function() {
      const file = this.files[0];
      if (file) {
        const reader = new FileReader();
        reader.onload = function(e) {
          base64Frame1Image = e.target.result;
          const previewElement = document.getElementById('frame1-preview');
          previewElement.src = base64Frame1Image;
          previewElement.style.display = 'block';
          systemLog(`ASSET MOUNTED: Target Image loaded into VRAM buffer.`);
        };
        reader.readAsDataURL(file);
      }
    });

    document.getElementById('action-trigger').addEventListener('click', async () => {
      systemLog('--- GENERATION SEQUENCE INITIATED ---');
      
      const stateSnapshot = {
        prompt: document.getElementById('prompt').value || "Abstract algorithmic entity",
        negative: document.getElementById('negative').value,
        dimensions: document.getElementById('size').value,
        activeFrame: parseInt(document.getElementById('timeline').value),
        rigProfile: document.getElementById('rig-profile').value,
        motionPreset: document.getElementById('motion-preset').value,
        frame1Image: base64Frame1Image,
        fxRigWeight: parseInt(document.getElementById('rig-weight').value) / 100,
        fxViscosity: parseInt(document.getElementById('fx-viscosity').value) / 100,
        fxMorph: parseInt(document.getElementById('fx-morph').value) / 100
      };

      let pipelineData = stateSnapshot;
      
      {
        const sortedKeys = Object.keys(localStorage).filter(k => k.startsWith("isg_")).sort();
        for (const moduleKey of sortedKeys) {
          try {
            const rawSource = localStorage.getItem(moduleKey);
            const binaryBlob = new Blob([rawSource], { type: 'text/javascript' });
            const executableModule = await import(URL.createObjectURL(binaryBlob));
            const pluginInstance = new executableModule.default();
            pipelineData = await pluginInstance.apply(999, { terminal: systemLog, lastOutput: pipelineData });
          } catch(err) {
            systemLog(`<span style="color:red">ERR [${moduleKey}]: ${err.message}</span>`);
          }
        }
      }
      
      systemLog(`Payload parsing complete.`);
      const outputDisplay = document.getElementById('final-payload-output');
      outputDisplay.value = typeof pipelineData === 'string' ? pipelineData : JSON.stringify(pipelineData, null, 2);
      
      // Auto-jump to the Oracle tab to see the output
      navigateToTab('oracle');
    });

    document.getElementById('file-loader').addEventListener('change', function() {
      for (const file of this.files) {
        file.text().then(content => {
          const key = "isg_" + file.name.replace('.js.txt', '').replace('.js', '');
          localStorage.setItem(key, content);
          systemLog(`Compiled Module: [${key}]`);
        });
      }
    });
  </script>
