  <script>
    const logBox = document.getElementById('log');
    const systemLog = (msg) => { logBox.innerHTML = `<div>> ${msg}</div>`; logBox.scrollTop = logBox.scrollHeight; };

    // Tab Router Engine
    const TabRegistry = ['core', 'perf', 'rig', 'canvas', 'oracle'];
    function navigateToTab(targetId) {
      document.querySelectorAll('.plugin-panel').forEach(panel => panel.classList.remove('active'));
      document.getElementById(`tab-${targetId}`).classList.add('active');
      document.querySelectorAll('.nav-btn').forEach(btn => btn.classList.remove('active'));
      document.getElementById(`btn-${targetId}`).classList.add('active');
    }
    TabRegistry.forEach(id => { document.getElementById(`btn-${id}`).addEventListener('click', () => navigateToTab(id)); });

    window.globalSensors = { abletonXScale: 0, bpmLocked: 148, shakerForce: 0 };
    let playheadInterval;
    let cachedFrames = [];

    // ABLETON PAD GESTURE ENGINE WITH INERTIA & MIRROR UPDATE
    const abletonSurface = document.getElementById('ableton-surface');
    const mainViewport = document.getElementById('fast-preview-img');
    const mirrorViewport = document.getElementById('mini-mirror-img');
    const mirrorHud = document.getElementById('mirror-hud');
    
    let isDragging = false, lastX, velocity = 0;

    abletonSurface.addEventListener('pointerdown', (e) => { 
      isDragging = true; 
      lastX = e.clientX;
      abletonSurface.classList.add('active'); 
      clearInterval(playheadInterval); // Loop anhalten bei Interaktion
    });

    window.addEventListener('pointermove', (e) => {
      if (!isDragging) return;
      const deltaX = e.clientX - lastX;
      velocity = deltaX * 0.25; // Impuls berechnen
      window.globalSensors.abletonXScale += deltaX * 0.04;
      
      document.getElementById('ableton-stats').innerText = `X-Pos: ${window.globalSensors.abletonXScale.toFixed(2)} | Inertia: ${velocity.toFixed(2)}`;
      
      if (cachedFrames.length === 12) {
        let currentFrame = Math.abs(Math.floor(window.globalSensors.abletonXScale)) % 12;
        mainViewport.src = cachedFrames[currentFrame];
        mirrorViewport.src = cachedFrames[currentFrame];
        document.getElementById('renderer-hud').innerHTML = `LIVE SCRUBBING<br>FRAME: ${currentFrame + 1}/12`;
      }
      lastX = e.clientX;
    });

    window.addEventListener('pointerup', () => {
      isDragging = false;
      abletonSurface.classList.remove('active');
      
      // Trägheits-Auslauf (Physics Friction Curve)
      let decay = setInterval(() => {
        if (Math.abs(velocity) < 0.08) { 
          clearInterval(decay); 
          if(cachedFrames.length === 12) startTwelveFpsPlayer(cachedFrames, document.getElementById('seed-val').value); 
          return; 
        }
        velocity *= 0.86; // Friction Koeffizient
        window.globalSensors.abletonXScale += velocity;
        document.getElementById('ableton-stats').innerText = `X-Pos: ${window.globalSensors.abletonXScale.toFixed(2)} | Inertia: ${velocity.toFixed(2)}`;
        
        if (cachedFrames.length === 12) {
          let currentFrame = Math.abs(Math.floor(window.globalSensors.abletonXScale)) % 12;
          mainViewport.src = cachedFrames[currentFrame];
          mirrorViewport.src = cachedFrames[currentFrame];
        }
      }, 25);
    });

    // EI-SHAKER ACCELEROMETER INTEGRATION WITH BRIGHT DOPAMINE FLASH
    document.getElementById('request-gyro').addEventListener('click', async () => {
      if (typeof DeviceMotionEvent !== 'undefined' && typeof DeviceMotionEvent.requestPermission === 'function') {
        try {
          let perm = await DeviceMotionEvent.requestPermission();
          if (perm === 'granted') initShaker();
        } catch(err) { systemLog("Sensor Block: " + err.message); }
      } else { initShaker(); }
    });

    function initShaker() {
      systemLog("EGG-SHAKER SENSORS UPLINKED. Start swinging your phone.");
      const hudBox = document.getElementById('shaker-hud-box');
      window.addEventListener('devicemotion', (e) => {
        const acc = e.accelerationIncludingGravity;
        if (!acc) return;
        const force = Math.abs(acc.x) + Math.abs(acc.y) + Math.abs(acc.z) - 9.8;
        if (force > 7.5) {
          window.globalSensors.shakerForce = force;
          document.getElementById('shaker-val').innerText = force.toFixed(1);
          
          hudBox.classList.add('flash'); // Kick Drum Trigger Flash Effekt
          setTimeout(() => hudBox.classList.remove('flash'), 75);
          
          const msPerBeat = (60 / window.globalSensors.bpmLocked) * 1000;
          const onGrid = (performance.now() % msPerBeat) < (msPerBeat / 4);
          document.getElementById('snap-status').innerText = onGrid ? "GRID SYNC: LOCKED" : "GRID SYNC: DRIFTING";
        }
      });
    }

    // MULTI-FRAME INFERENCE ENGINE (12 FRAMES PARALLEL)
    document.getElementById('action-trigger').addEventListener('click', async () => {
      systemLog('--- COMPILING LATENT MATRIX GENERATION ---');
      
      const promptValue = document.getElementById('prompt').value;
      const isSeedLocked = document.getElementById('use-seedlock').checked;
      const masterSeed = isSeedLocked ? document.getElementById('seed-val').value : Math.floor(Math.random() * 89999) + 10000;
      if (!isSeedLocked) document.getElementById('seed-val').value = masterSeed;
      
      document.getElementById('renderer-hud').innerHTML = `BURSTING 12 FRAMES...<br>SEED: ${masterSeed}`;
      
      cachedFrames = [];
      let loadedCount = 0;
      mirrorHud.style.display = 'none';

      for (let i = 0; i < 12; i++) {
        let neg = "blurry, 13-point rig";
        if (document.getElementById('block-vocals').checked) neg += ", vocals, singing, human voice track";
        if (document.getElementById('block-human-voice').checked) neg += ", talking, podcast background";

        const dynamicPrompt = `${promptValue}, sequence step ${i} of 12, mechanical step movement trajectory, high contrast raw vector ink, dirty industrial cartoon style`;
        const url = `https://image.pollinations.ai/p/${encodeURIComponent(dynamicPrompt)}?width=512&height=512&seed=${masterSeed}&nologo=true`;
        
        const imgObj = new Image();
        imgObj.src = url;
        imgObj.onload = () => {
          loadedCount++;
          cachedFrames[i] = url;
          document.getElementById('renderer-hud').innerHTML = `RESOLVING LATENT GRID: ${loadedCount}/12<br>SEED: ${masterSeed}`;
          
          if (loadedCount === 12) {
            systemLog("12-Frame Sequence im RAM gefangen. Starte 12fps Loop.");
            mirrorViewport.src = cachedFrames[0];
            mirrorViewport.style.display = 'block';
            startTwelveFpsPlayer(cachedFrames, masterSeed);
          }
        };
      }

      // Baue Log-Daten zusammen
      document.getElementById('final-payload-output').value = `[MASTER PAYLOAD]\nPrompt: ${promptValue}\nLocked Seed: ${masterSeed}\nFilters: Active Mutes\nShaker Amplitude: ${window.globalSensors.shakerForce.toFixed(2)} G\nTarget FPS: 12`;
      navigateToTab('canvas');
    });

    function startTwelveFpsPlayer(frames, seed) {
      clearInterval(playheadInterval);
      let current = 0;
      playheadInterval = setInterval(() => {
        mainViewport.src = frames[current];
        document.getElementById('renderer-hud').innerHTML = `PLAYBACK ACTIVE (12FPS)<br>FRAME: ${current + 1}/12 | SEED: ${seed}`;
        current = (current + 1) % 12;
      }, 83); // Loop läuft exakt auf 12 Bildern pro Sekunde
    }

    function copyPayload() {
      document.getElementById('final-payload-output').select();
      navigator.clipboard.writeText(document.getElementById('final-payload-output').value);
      systemLog("Logs to Clipboard pushed.");
    }
  </script>
