export default class IsolatedPlugin {
  constructor() {
    this.metadata = { name: "Plugin Name", id: "unique_id" };
  }

  async apply(seed, context) {
    // 1. Read what the PREVIOUS module calculated
    const pipeIn = context.lastOutput; 
    
    // 2. Perform isolated math or manipulation
    const pipeOut = doSomething(pipeIn, seed); 
    
    // 3. Print to the shared terminal screen
    context.terminal("Log message here"); 
    
    // 4. Pass the modified data packet to the NEXT module
    return pipeOut; 
  }
}
